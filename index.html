⚙️ Elevation Browser 设置

首次使用需要配置数据来源。请选择一种方式：

方式一：自动获取
方式二：手动粘贴

输入你的 Obsidian Publish Site Hash（在 Publish 站点按 F12 → 网络面板中查找）：

💡 在你的 Publish 网站按 F12 → 网络(Network) → 找到包含 "publish-01.obsidian.md/access/" 的请求 → 复制 access/ 后面的那串字符

在你的 Obsidian Publish 网站的浏览器控制台(F12 → Console)中运行以下脚本：

// 请先在上方"自动获取"中找到 site hash，然后替换下面的值 const HASH = '在这里粘贴你的site_hash'; (async()=>{const B=`https://publish-01.obsidian.md/access/${HASH}`;const c=await fetch(`https://publish-01.obsidian.md/cache/${HASH}`,{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({id:HASH,version:'1'})});const idx=await c.json();const lk={};Object.keys(idx).forEach(p=>{lk[p.split('/').pop().toLowerCase()]=p});const ef=Object.keys(idx).filter(p=>/06 Data\/Elevation\/A-\d+\.md/i.test(p)).sort();const R=[];for(const f of ef){try{const r=await fetch(`${B}/${f.split('/').map(encodeURIComponent).join('/')}`);const md=await r.text();const imgs=[];const seen=new Set();let m;const r1=/!\[\[([^\]|]+\.(jpg|jpeg|png|gif|webp|svg|bmp))(\|[^\]]*)?\]\]/gi;while(m=r1.exec(md)){let p=m[1].trim();if(!p.includes('/')){const rp=lk[p.toLowerCase()];if(rp)p=rp}const u=`${B}/${p.split('/').map(encodeURIComponent).join('/')}`;if(!seen.has(u)){seen.add(u);imgs.push(u)}}const r2=/!\[[^\]]*\]\(([^)\s]+\.(jpg|jpeg|png|gif|webp|svg|bmp))\)/gi;while(m=r2.exec(md)){let u=m[1];if(!u.startsWith('http')){let p=u;if(!p.includes('/')){const rp=lk[p.toLowerCase()];if(rp)p=rp}u=`${B}/${p.split('/').map(encodeURIComponent).join('/')}`}if(!seen.has(u)){seen.add(u);imgs.push(u)}}const id=f.match(/A-(\d+)/);R.push({id:id?`A-${id[1]}`:f,title:`Elevation ${id?`A-${id[1]}`:f}`,tags:[],images:imgs});console.log(`✅ ${id?`A-${id[1]}`:f}: ${imgs.length} images`)}catch(e){console.error(`❌ ${f}:`,e)}}const out=JSON.stringify(R);try{await navigator.clipboard.writeText(out);console.log('📋 已复制到剪贴板！')}catch(e){}console.log(out)})();
